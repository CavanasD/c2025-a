## Task: Reverse Polish notation Calculato
```C++
//Reverse Polish Calculation  
#include<iostream>  
#include<stack>  
#include<string>  
#include<sstream>  
  
using namespace std;  
int main() {  
    stack<int> st;  
    string CalculateStr, ExpressionALL;  
    cout << "Please enter expression: " << endl;  
    getline(cin, ExpressionALL);  
    stringstream ss(ExpressionALL);  
    while (ss >> CalculateStr) {  
        if (isdigit(CalculateStr[0])) {  
            st.push(stoi(CalculateStr));  //数字入栈  
        } else {  
            int b = st.top(); st.pop();  
            int a = st.top(); st.pop();  
            //下面运算  
            if (CalculateStr == "+") st.push(a + b);  
            else if (CalculateStr == "-") st.push(a - b);  
            else if (CalculateStr == "*") st.push(a * b);  
            else if (CalculateStr == "/") st.push(a / b);  
        }  
    }  
    cout << st.top() << endl;  
    return 0;  
}
```
