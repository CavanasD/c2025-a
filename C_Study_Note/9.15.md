```C++
#include<iostream>  
  
using namespace std;  
  
int queue[100];  
int firstp, endp = 0;  
  
void inQueue(int x) {  
    if (endp == 100) {  
        cout << "Warning: full queue" << endl;  
        return;  
    }  
    queue[endp++] = x;  
}  
int outQueue() {  
    if (firstp == endp) {  
        cout << "Warning: empty queue!" << endl;  
        return -1;  
    }  
    return queue[firstp++];  
}  
  
int main() {  
    //queue in  
    inQueue(1);  
    inQueue(2);  
    int x = outQueue();  
    cout << x << endl;  
    x = outQueue();  
    cout << x << endl;  
    return 0;  
}
```

```C
    int i = 0;  
    while (1) {  
        if (queue[i] == 0) {  
            queue[i] = number;  
            return;  
        } else {  
            i++;  
        }  
    }  
}  
  
int outQueue() {  
    int j = queue[0], i = 0;  
    do {  
        queue[i] = queue[i + 1];  
        i++;  
    } while (queue[i] == 0);  
    return j;  
}

    inQueue(1);  
    inQueue(2);  
    int x = outQueue();  
    assert(x==2);
```

```C
#include<stdio.h>  
#include<assert.h>  
  
typedef struct {  
    int *data;  
    int front,count;  
    int capacity;  
}Queue;  
  
Queue* createQueue(int capacity);  
Queue* createQueue(int capacity) {  
    Queue* q = (Queue*)malloc(sizeof(Queue));  
    q->data = (int*)malloc(capacity * sizeof(int));  
    q->capacity = capacity;  
    q->front = 0;  
    q->count = 0;  
    //q is queue  
    return q;  
}  
void push(Queue* q, int value);  
void push(Queue* q, int value) {  
    q->data[q->front+q->count++] = value;  
}  
  
int pop(Queue* q) ;  
int pop(Queue* q) {  
    assert(q->count != NULL);  
    int value = q->data[q->front++];  
    q->count--;  
    return value;  
}  
  
int main() {  
    Queue* q1 = createQueue(10);  
    assert(q1 != NULL);  
    assert(q1->capacity >0);  
    assert(q1->data != NULL);  
  
    push(q1, 1);  
    push(q1, 2);  
  
}
```
